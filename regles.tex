\section{Définition}

\subsection{Représentation du train}
Dans notre modèle, un train est représenté par un quintuplet $(id, pos, dir, P, rel)$ où :
\begin{itemize}
    \item $id$ : identifiant du train
    \item $pos$ : position du train 
    \item $dir$ : direction du train
    \item $P$ : programme du train
    \item $rel$ : position relative du train par rapport au départ
\end{itemize}

\subsection{Représentation du régulateur}
Le régulateur est représenté par un septuplet $(E, J, S, W, G, F, B)$ où :
\begin{itemize}
    \item $E$ : événements du régulateur
    \item $J$ : tableau de jetons
    \item $S$ : tableau des aiguillages
    \item $W$ : file d'attente
    \item $G$ : garde
    \item $F$ : feux de signalisation
    \item $B$ : buffer (unidirectionnel) de messages
\end{itemize}

\subsection{Définition des prédicats et fonctions}

\begin{itemize}
    \item $suiv(pos, dir, switch)$ : renvoie la position suivante 
    \item $handleTL(id, {\color{Red}pos, dir}, S)$ : renvoie la nouvelle valeur des feux de signalisation 
\end{itemize}

\subsection{Actions sur le train}

\noindent

Start : 
\inferrule
    { dir \neq D \\ R.G = false}
    {\trainfull{id}{pos}{dir}{\concat{SU(D,[\concat{n}{N'}])}{P'}}{rel}, R \\ \xRightarrow{\text{!ev(id)}} \\ \trainfull{id}{pos}{D}{\concat{SU(D,[\concat{N}{N'}])}{P'}}{rel},R}
\vspace{0.5cm}


Stop :
\inferrule
    { P = \varepsilon \\ dir \neq * \\ R.G = false}
    {\trainfull{id}{pos}{dir}{P}{rel},R \\ \xRightarrow{\text{!ev(id)}} \\ \trainfull{id}{pos}{*}{P}{rel},R}
\vspace{0.5cm}


Until :
    \inferrule
    { suiv(pos, D, S) = n \\ {\color{PineGreen} R.F[(pos,dir)] = V} \\ {\color{PineGreen}R.G = false}}
    {\trainfull{id}{pos}{D}{\concat{SU(D,[\concat{n}{N'}])}{P'}}{rel},R \\ \xRightarrow{\text{!ev(id)}} \\ \trainfull{id}{n}{D}{\concat{SU(D,[N'])}{P'}}{rel+1},R}
\vspace{0.5cm}


Until$_{cons}$ :
    \inferrule
        { suiv(pos, D, S) = n \\ {\color{PineGreen} R.F[(pos,dir)] = V} \\ {\color{PineGreen} R.G = false}}
        {\trainfull{id}{pos}{D}{\concat{SU(D,[\concat{n}{\varepsilon}])}{P'}}{rel},R\\ \Rightarrow \\ \trainfull{id}{n}{D}{P'}{rel+1},R}
\vspace{0.5cm}

\subsection{Actions sur le régulateur : simulation grands pas avec une approche petits pas}

\noindent

Recv :
\inferrule
    { len(B) = x}
    {\reg{E}{J}{S}{W}{G}{F}{B} \\ \xRightarrow{\text{?ev(id)}} \\ \reg{E}{J}{S}{W}{G}{F}{\majtab{B}{x+1}{id}}}
\vspace{0.5cm}

StarEvent :
\inferrule
    { len(B) \neq 0}
    {\reg{E}{J}{S}{W}{false}{F}{B} \\ \Rightarrow \\ \reg{E}{J}{S}{W}{true}{F}{B}}
\vspace{0.5cm}

Incr$_{bf}$ : %incr qui arrive avant un att
\inferrule
    { Head(B) = t_{id} \\ J[j] = x \\ W[(j, x+1)] = \varepsilon }
    {\reg{E[t_{id}] \owns \concat{incr(j)}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{\majtab{J}{j}{x+1}}{S}{W}{true}{F}{B}}
\vspace{0.5cm}

Incr$_{af}$ : %incr qui arrive après un att
\inferrule
    { Head(B) = t_{id} \\ J[j] = x \\ W[(j, x+1)] = w_{id} \\ handleTL(w_{id}, {\color{Red}pos, dir}, S) = F'}
    {\reg{E[t_{id}] \owns \concat{incr(j)}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{\majtab{J}{j}{x+1}}{S}{W}{true}{F'}{B}}
\vspace{0.5cm}

Att$_{bf}$ : %att qui arrive avant un incr
\inferrule
    { Head(B) = t_{id} \\ J[j] = x \\ x \neq y}
    {\reg{E[t_{id}] \owns \concat{att(j,y)}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{J}{S}{{\majtab{W}{(j,y)}{id}}}{true}{F}{B}}
\vspace{0.5cm}

Att$_{af}$ : %att qui arrive après un incr
\inferrule
    { Head(B) = t_{id} \\ J[j] = x \\ handleTL(id, {\color{Red}pos, dir}, S) = F'}
    {\reg{E[t_{id}] \owns \concat{att(j,x)}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{J}{S}{W}{true}{F'}{B}}
\vspace{0.5cm}

Turn : 
\inferrule
    { Head(B) = t_{id}}
    {\reg{E[t_{id}] \owns \concat{turn(id_{s},state)}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{J}{\majtab{S}{id_{s}}{state}}{W}{true}{F}{B}}
\vspace{0.5cm}

Auth : 
\inferrule
    { Head(B) = t_{id} \\ handleTL(id, {\color{Red}pos, dir}, S) = F'}
    {\reg{E[t_{id}] \owns \concat{auth}{P'}}{J}{S}{W}{true}{F}{B} \\ \Rightarrow \\ \reg{\majtab{E}{t_{id}}{P'}}{J}{S}{W}{true}{F'}{B}}
\vspace{0.5cm}

EndEvent :
\inferrule
    { Head(B) = t_{id} \\ E[t_{id}] = \varepsilon}
    {\reg{E}{J}{S}{W}{true}{F}{\concat{m}{B'}} \\ \Rightarrow \\ \reg{E}{J}{S}{W}{false}{F}{B'}}
\vspace{0.5cm}
