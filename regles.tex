\section{Définition}

\subsection{Représentation du train}
Dans notre modèle, un train est représenté par un quadruplet $(id, pos, dir, P)$ où :
\begin{itemize}
    \item $id$ : identifiant du train
    \item $pos$ : position du train 
    \item $dir$ : direction du train
    \item $P$ : programme du train
\end{itemize}

\subsection{Représentation du régulateur}
Le régulateur est représenté par un sextuplet $(J, A, E, N_e, W, S)$ où :
\begin{itemize}
    \item $J$ : tableau de jetons
    \item $A$ : tableau des autorisations
    \item $E$ : événements du régulateur
    \item $N_e$ : tableau des prochains événements
    \item $W$ : file d'attente
    \item $S$ : tableau des aiguillages
\end{itemize}


\subsection{Notation}
On note
\\
$\mathbf{O}_{e}$ l'ensemble des ordres des événements
\vspace{0.3cm} \\
dict : $\mathbf{A} \rightarrow \mathbf{B}$ avec $\mathbf{A}$ l'ensemble des clés et $\mathbf{B}$ l'ensemble des valeurs
\\ $\mathbf{A} \times \mathbf{B}$ telque \{ $\langle k,v \rangle$ si $k_{j} \neq k_{i}$ \} \\
\vspace{0.1cm} \\
suppr(D,k) = \bracket{D \backslash \{\langle x,D[x] \rangle\} \text{ si } k = x}{D \text{ sinon}}
\\
\vspace{0.3cm} \\
E : dict avec \bracket{a \in \mathbf{A} | a = \langle id_{t}, numEv \rangle}{\mathbf{B} \subseteq \mathbf{O}_{e}}
\vspace{0.3cm} \\
W : dict avec \bracket{a \in \mathbf{A} | a = \langle id_{c}, val_{j} \rangle}{\mathbf{B} \subseteq \mathbf{N}}
\vspace{0.3cm} \\
Soit un tableau T, on note la mise à jour de T à la position i avec la valeur v par $\majtab{T}{i}{v}$ 

\subsection{Définition des prédicats et fonctions}

\begin{itemize}
    \item $suiv(pos, dir, switch)$ : renvoi la position suivante 
\end{itemize}

\section{Règles}

\subsection{Actions sur le train}

Start : 
\inferrule
    { dir \neq D}
    {\trainfull{id}{pos}{dir}{\concat{SU(D,N)}{P'}}, R \Rightarrow \trainfull{id}{pos}{D}{\concat{SU(D,N)}{P'}}, R} 
\vspace{0.5cm}


Stop :
\inferrule
    { P = \varepsilon \\ dir \neq * \\ (id, Ne[id]) \notin E}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{*}{P}, \reg{J}{E}{Ne}{A}{W}{S}}
\vspace{0.5cm}


Until :
    \inferrule
    { suiv(pos, D, S) = pos' \\ pos' \neq N \\ pos' \neq \varepsilon \\ A[id] = y \\ y \neq 0 \\ (id, x) \notin E \\ Ne[id] = x}
    {\trainfull{id}{pos}{D}{\concat{SU(D,N)}{P'}}, \reg{J}{E}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos'}{D}{\concat{SU(D,N)}{P'}}, \reg{J}{E}{\majtab{Ne}{id}{x+1}}{\majtab{A}{id}{y-1}}{W}{S}}
\vspace{0.5cm}


Until' :
    \inferrule
        { suiv(pos, D, S) = pos' \\ A[id] = y \\ y \neq 0 \\(id, x) \notin E \\ Ne[id] = x}
        {\trainfull{id}{pos}{D}{\concat{SU(D,pos')}{P'}}, \reg{J}{E}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos'}{D}{P'}, \reg{J}{E}{\majtab{Ne}{id}{x+1}}{\majtab{A}{id}{y-1}}{W}{S}}
\vspace{0.5cm}

\subsection{Actions sur le régulateur}

elim$_{ev}$ :
\inferrule
    { E[ev] = \varepsilon \\ ev = (id, Ne[id])}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P}, \reg{J}{\supprdict{E}{ev}}{Ne}{A}{W}{S}}
\vspace{0.5cm}

incr$_{bf}$ : %incr qui arrive avant un att
\inferrule
    {  ev = (id, Ne[id]) \\ J[j] = x \\ W[(j, x+1)] = \varepsilon }
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = \concat{incr(j)}{P'}}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P}, \reg{\majtab{J}{j}{x+1}}{E[ev] = P'}{Ne}{A}{W}{S}}
\vspace{0.5cm}

incr$_{af}$ : %incr qui arrive après un att
\inferrule
    {  ev = (id, Ne[id]) \\ J[j] = x \\ W[(j, x+1)] = w_{id} \\ nextAtt(W_{id},R) = num}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = \concat{incr(j)}{P'}}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P} \\ \reg{\majtab{J}{j}{x+1}}{E[ev] = P'}{Ne}{\majtab{A}{w_{id}}{num}}{\supprdict{W}{(j, x+1)}}{S}}
\vspace{0.5cm}

att$_{bf}$ : %att avant que incr soit appliqué
\inferrule
    {  ev = (id, Ne[id]) \\ J[j] = x \\ x \neq y}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = \concat{att(j, y)}{P'}}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = P'}{Ne}{A}{\majtab{W}{(j,y)}{id}}{S}}
\vspace{0.5cm}

att$_{af}$ : %att avant que incr soit appliqué
\inferrule
    {  ev = (id, Ne[id]) \\ J[j] = x \\ nextAtt(id,R) = num}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = \concat{att(j, x)}{P'}}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = P'}{Ne}{\majtab{A}{id}{num}}{W}{S}}
\vspace{0.5cm}

turn :
\inferrule
    {ev = (id, Ne[id])}
    {\trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = \concat{turn(id_{s},state)}{P'}}{Ne}{A}{W}{S} \\ \Rightarrow \trainfull{id}{pos}{dir}{P}, \reg{J}{E[ev] = P'}{Ne}{A}{W}{\majtab{S}{id_{s}}{state}}}
